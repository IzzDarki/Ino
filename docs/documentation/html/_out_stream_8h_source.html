229  {
230  DecimalsRound ReturnVal = { 0, false };
231  uint8_t Zeros = 0;
232  uint8_t Nines = 0;
233  Num = abs(Num) - GetDecimalPart(Num);
234  while (Zeros < 2 && Nines < 2 && ReturnVal.Decimals < 6)
235  {
236  Num *= 10;
237  if (static_cast<int>(Num) == 0)
238  {
239  Zeros++;
240  ReturnVal.Decimals += Nines;
241  Nines = 0;
242  }
243  else if (static_cast<int>(Num) == 9)
244  {
245  Nines++;
246  ReturnVal.Decimals += Zeros;
247  Zeros = 0;
248  Num -= 9;
249  }
250  else {
251  ReturnVal.Decimals++;
252  ReturnVal.Decimals += (Zeros + Nines);
253  Zeros = 0;
254  Nines = 0;
255  Num -= static_cast<int>(Num);
256  }
257  }
258  if (Zeros && static_cast<int>(Num *= 10) == 9)
259  {
260  ReturnVal.Decimals += Zeros;
261  ReturnVal.Round = true;
262  }
263  if (Nines)
264  ReturnVal.Round = true;
